As-Patterns

isSubSeqOf :: Eq a => [a] -> [a] -> Bool
isSubSeqOf sub full = go sub full
  where go [] _ = True
        go _ [] = False
        go subrem@(x:xs) (y:ys) =
          if x == y then go xs ys
          else go subrem ys

As-Patterns

capitalizeWords :: String -> [(String, String)]
capitalizeWords sentence = map capitalize (words sentence)
  where capitalize "" = ("", "")
        capitalize w@(x:xs) = (w, toUpper x : xs)
